Function URLEncode(s As String) As String
    Dim i As Long, c As String, ac As Integer
    For i = 1 To Len(s)
        c = Mid(s, i, 1)
        ac = Asc(c)
        If (ac >= 48 And ac <= 57) Or (ac >= 65 And ac <= 90) Or (ac >= 97 And ac <= 122) Or ac = 45 Or ac = 95 Or ac = 46 Or ac = 126 Then
            URLEncode = URLEncode & c
        Else
            URLEncode = URLEncode & "%" & Right("0" & Hex(ac), 2)
        End If
    Next i
End Function

End Function

Function RunSilent(exePath As String, arguments As String) As Boolean
    On Error GoTo Fail
    Dim sei As SHELLEXECUTEINFO
    sei.cbSize = Len(sei)
    sei.fMask = SEE_MASK_NOCLOSEPROCESS
    sei.lpVerb = StrPtr("open")
    sei.lpFile = StrPtr(exePath)
    sei.lpParameters = StrPtr(arguments)
    sei.nShow = 0
    If ShellExecuteExW(sei) Then
        If sei.hProcess Then
            WaitForSingleObject sei.hProcess, 30000
            CloseHandle sei.hProcess
            RunSilent = True
            Exit Function
        End If
    End If
Fail:
    RunSilent = False
End Function

Function MemMem(haystack() As Byte, needle As String) As Long
    Dim i As Long, j As Long, found As Boolean
    Dim needleBytes() As Byte: needleBytes = StrConv(needle, vbFromUnicode)
    For i = 0 To UBound(haystack) - UBound(needleBytes)
        found = True
        For j = 0 To UBound(needleBytes)
            If haystack(i + j) <> needleBytes(j) Then
                found = False
                Exit For
            End If
        Next j
        If found Then
            MemMem = i
            Exit Function
        End If
    Next i
    MemMem = -1
End Function
